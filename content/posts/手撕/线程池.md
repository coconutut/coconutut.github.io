C++实现线程池

```c++
#include <iostream>
#include <thread>
#include <condition_variable>
#include <vector>
#include <functional>
#include <mutex>
#include <queue>
#include <memory>
#include <future>
#include <type_traits>

class threadPool{
public:
    threadPool(size_t thread_num){
        for(int i = 0; i < thread_num; i++){
            threads_.emplace_back([this](){
                this->worker();
            });
        }
    }
    
    ~threadPool(){
        stopFlag = true;
        cv_.notify_all(); //唤醒所有线程
        for(auto& t : threads_){
            t.join();
        }
    }
    
    template <typename F, typename... Args>
    auto pushTask(F&& f, Args&&... args)
        -> std::future<typename std::invoke_result<F, Args...>::type>
    {
        using Ret = typename std::invoke_result<F, Args...>::type;

        //把参数打包成无参任务
        auto task = std::make_shared<std::packaged_task<Ret()>>(
            //完美转发
            std::bind(std::forward<F>(f), std::forward<Args>(args)...)
        );

        std::future<Ret> res = task->get_future();

        {
            std::lock_guard<std::mutex> lock(mtx_);
            tasks_.emplace([task](){ (*task)(); });
        }

        cv_.notify_one();

        return res;
    }

private:

    void worker(){
        while(true){
            std::function<void()> task;
            {
                std::unique_lock<std::mutex> lock(mtx_);
                cv_.wait(lock, [this](){return stopFlag || !tasks_.empty();});
                if(stopFlag && tasks_.empty()) return;
                task = tasks_.front();
                tasks_.pop();
            }
            task();
        }
    }

private:
    bool stopFlag{false};
    std::vector<std::thread> threads_;
    std::queue<std::function<void()>> tasks_;
    std::condition_variable cv_;
    std::mutex mtx_;

    //拷贝构造
    threadPool(const threadPool&) = delete;
    //移动构造
    threadPool(threadPool&&) = delete; //移动需要改变源对象，不能使用const修饰

    //拷贝赋值
    threadPool& operator=(const threadPool&) = delete;
    //移动赋值
    threadPool& operator=(threadPool&&) = delete;
};

int main(){
    threadPool pool(5);

    std::future<int> res = pool.pushTask([](int a, int b){
        return a + b;
    }, 1, 1);

    std::cout << res.get() << std::endl;
    
    return 0;
}
```
