1. 哈希：通过哈希函数建立关键字与存储位置的映射关系
2. 哈希冲突：不同关键字通过哈希函数计算出相同的存储位置
3. 负载因子：哈希表元素大小 / 哈希表大小
4. 哈希函数目标：让关键字均匀散列到哈希表中，减少冲突
5. 常用哈希函数：除数散列法、乘法散列法、全域散列法
6. 哈希冲突解决方案： 
  * 开放定址法（线性探测、二次探测）
  * 哈希桶（哈希表存链表指针，冲突元素挂在对应位置的链表上，负载因子可 > 1）
7. 哈希桶手撕代码
```c++
#include <vector>
#include <cstddef>
#include <string>
#include <iostream>
using namespace std;

//链表节点
template<class K, class V>
struct node{
        K key;
        V value;
        node* next;
        node(K k, V v) : key(k), value(v), next(nullptr) {};
};
                
template<class K, class V>
class HashTable{
public:
        //构造函数
        HashTable(size_t cap = 8) : size_(0), buckets_(cap, nullptr) {}

        //析构函数
        ~HashTable(){
                for(auto head : buckets_){
                        while(head){
                                node<K, V>* next = head->next;
                                delete head;
                                head = next;
                        }
                }
        }

        //查找
        node<K, V>* find(K key){
                size_t idx = hash(key) % buckets_.size();
                node<K, V>* cur = buckets_[idx];
                while(cur){
                        if(cur->key == key) return cur;
                        cur = cur->next;
                }
                return nullptr;
        }

        //插入
        bool insert(K key, V value){
                size_t idx = hash(key) % buckets_.size(); //除法散列法
                node<K, V>* cur = buckets_[idx];
                //查重
                while(cur){
                        if(cur->key == key) return false;
                        cur = cur->next;
                }
                //头插法
                node<K, V>* newNode = new node(key, value);
                newNode->next = buckets_[idx];
                buckets_[idx] = newNode;
                size_++;
                return true;
        }

        //删除
        bool erase(K key){
                size_t idx = hash(key) % buckets_.size();
                node<K, V>* cur = buckets_[idx];
                node<K, V>* prev = nullptr;
                while(cur){
                        if(cur->key == key){
                                if(prev == nullptr) buckets_[idx] = cur->next;
                                else prev->next = cur->next;
                                delete cur;
                                --size_;
                                return true;
                        }
                        prev = cur;
                        cur = cur->next;
                }
                return false;
        }

        //获取大小
        size_t size(){
                return size_;
        }

        //判空
        bool empty(){
                return size_ == 0;
        }

private:
        //哈希值转换
        size_t hash(const K& key){
                if constexpr(is_same<K, string>::value){
                    size_t ans = 0;
                    for(char c : key){
                        ans += ans*131 + c;
                    }
                    return ans;
                }
                else{
                    return static_cast<size_t>(key);
                }
        }
        
        //哈希桶
        vector<node<K, V>*> buckets_;
        
        //哈希表元素大小
        size_t size_;
};

int main(){
    HashTable<int, int> mp;
    mp.insert(1, 1);
    mp.insert(2, 2);
    cout << mp.size() << endl;
    mp.erase(1);
    cout << mp.find(2)->value << endl;;
    cout << mp.size() << endl;
}
