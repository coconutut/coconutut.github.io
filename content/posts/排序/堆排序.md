1. 堆是完全二叉树的结构
2. 二叉树的性质：
    * 符合数组顺序存储，其余层满，最后一层从左到右，不可空缺
    * 索引`i`的左子节点索引为`2*(i+1)-1`，右子节点索引为`2*(i+1)`，父节点索引为`(i-1)/2`
    * 节点数量为`n`的二叉堆最后一个非叶节点索引为`n/2-1`
    * 一个高度为`h`的完全二叉树的节点数量介于高度`h-1`的满二叉树和高度`h`的满二叉树之间`2^(h-1)-1 < n < 2^h-1`，可推出节点数量为`n`的完全二叉树的高度为`(log2)(n+1)`（向上取整）
3. 手撕堆排序
```c++
#include <iostream>
#include <vector>
using namespace std;

class HeapSort {
public:
    // 构造函数，传入待排序数组
    HeapSort(vector<int>& arr) : nums(arr), size_(arr.size()) {}
    void sort() {
        //建最大堆顶
        for(int i = size_ / 2 - 1; i >= 0; i--){
            heapify(size_, i);
        }
        
        int len = size_;
        //交换堆顶与最后一个数
        while(len > 0){
            swap(nums[len-1], nums[0]);
            //更新长度
            len--;
            heapify(len, 0);
        }

        for(int n : nums){
            cout << n << " ";
        }
    }

private:
    // 维护最大堆
    // n--维护堆的长度 i--当前操作数组索引位置
    void heapify(int n, int i){
        int largestIndex = i;
        int l = 2*(i+1)-1, r = 2*(i+1);
        if(l < n && nums[l] > nums[largestIndex]) largestIndex = l;
        if(r < n && nums[r] > nums[largestIndex]) largestIndex = r;
        if(largestIndex != i){
            swap(nums[i], nums[largestIndex]);
            heapify(n, largestIndex);
        } 
    }

    vector<int>& nums; 
    size_t size_; //节点数量
};

//test
int main(){
    vector<int> nums{8, 777, 9, 5, 4, 66, 32};
    HeapSort instance(nums);
    instance.sort();
    return 0;
}
```
