1. 快速排序时间复杂度最优Onlogn，最差时间复杂度On2，空间复杂度O1
2. 填坑法
    * 先找最左侧元素当基准
    * 循环左侧指针小于右侧指针
    * 从右往左找小于基准的元素
    * 找到符合的存到左侧->右侧指针存储空
    * 从左往右找大于基准的元素
    * 找到符合的存到右侧->左侧指针存储空
    * 循环结束，填基准值
    * 递归两侧区间
```c++
#include <iostream>
#include <vector>
using namespace std;

void dfs(int left, int right, vector<int>& nums){
    //终止条件
    if(left >= right) return;

    int privot = nums[left];
    int l = left, r = right;
    while(l < r){
        //先从右往左找
        while(l < r && nums[r] >= privot) r--;
        nums[l] = nums[r]; //r空了（最初是因为l存储的是privot)
        //从左往右找
        while(l < r && nums[l] <= privot) l++;
        nums[r] = nums[l]; //l空了
    }
    //填坑
    nums[l] = privot;

    //递归
    dfs(left, l-1, nums);
    dfs(l+1, right, nums);
}

void quickSort(vector<int>& nums){
    int size = nums.size();
    dfs(0, size-1, nums);
}

int main(){
    vector<int> nums{4, 6, 7, 10, 19, 3, 999, 0};
    quickSort(nums);
    for(int n : nums){
        cout << n << " ";
    }
    return 0;
}
```
