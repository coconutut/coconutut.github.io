1. 冒泡排序时间复杂度`On2`，空间复杂度`O1`

```c++
#include <iostream>
#include <vector>
using namespace std;

void bubbleSort(vector<int>& nums){
    size_t size = nums.size(); //数组大小
    for(int i = size-1; i > 0; i--){  //j的边界
        for(int j = 0; j < i; j++){
            if(nums[j] > nums[j+1]){
                swap(nums[j], nums[j+1]);
            }
        }
    }
}

//test
int main(){
    vector<int> nums{22, 6, 6, 8, 8, 10, 0, 45, 1, 2};
    bubbleSort(nums);
    for(int n : nums){
        cout << n << " ";
    }
    return 0;
}
```
