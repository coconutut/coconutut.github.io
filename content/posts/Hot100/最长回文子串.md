1. 本文用易于理解的中心扩展法进行解题
```c++
class Solution{
public:
    string longestPalindrome(string s){
        size_t size = s.size();
        int begin = 0;
        int maxLen = 1;

        //扩展函数
        function<void(int, int)> expand = [&](int left, int right){
            while(left >= 0 && right < size && s[left] == s[right]){
                left--;
                right++;
            }

            //计算长度，每次都多扩展了一次
            int len = right - left - 1; //right - left + 1 - 2
            
            //判断是否比目前的长度大
            if(maxLen < len){
                begin = left + 1;
                maxLen = len;
            }
        };

        for(int i = 0; i < size; i++){
            expand(i, i);
            expand(i, i+1);
        }
        
        return s.substr(begin, maxLen);
    }
};
```
