1. 状态定义：`dp[i]`表示以`s[i-1]`结尾的有效括号长度
2. 转移方程:
    * 如果`s[i-1] = '('`时，`dp[i]`为0；
    * 如果`s[i-1] = ')'`时
        * 如果`s[i-2] = '('`，`dp[i] = dp[i-2] + 2`
        * 如果`s[i-2] = ')'`
            * 判断`s[i-1-dp[i-1]-1] ?= '('` -> `dp[i] = dp[i-1] + 2 + dp[i-dp[i-1]-2]`
```c++
class Solution {
public:
    int longestValidParentheses(string s) {
        size_t n = s.size();
        vector<int> dp(n+1, 0);

        int ans = 0;
        for(int i = 2; i <= n; i++){
            if(s[i-1] == ')'){
                if(s[i-2] == '('){
                    dp[i] = dp[i-2] + 2;
                }else{
                    if(i - dp[i-1] - 2 >= 0 && s[i - dp[i-1] - 2] == '('){
                        dp[i] = dp[i-1] + dp[i-dp[i-1]-2] + 2;
                    }
                }
            }
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
```
