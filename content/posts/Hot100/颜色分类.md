1. 经典荷兰三色国旗问题
2. 基于双指针的思路：
    * p代表下一个0要存放的位置
    * q代表下一个1要存放的位置
```c++
class Solution {
public:
    void sortColors(vector<int>& nums) {
        int p = 0, q = 0;
        for(int i = 0; i < nums.size(); ++i){
            if(nums[i] == 1){
                swap(nums[i], nums[q]);
                q++;
            }
            else if(nums[i] == 0){
                //碰到0了先换到1要换的位置q, 再把0换到p的位置, p的位置原先是1
                swap(nums[i], nums[q]);
                swap(nums[q], nums[p]);
                p++;
                q++;
            }
        }
    }
};
```
