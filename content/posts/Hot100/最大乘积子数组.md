1. 子数组是一定要连续的，要么叠加前一个状态，要么从当前状态开始
2. 乘积的最值要记录最大乘积与最小乘积

```c++
class Solution{
public:
    int maxProduct(vector<int>& nums){
        size_t n = nums.size();
        int dpmax = nums[0];
        int dpmin = nums[0];

        int ans = nums[0];

        for(int i = 1; i < n; i++){
            //当数为负数的时候，直接交换
            if(nums[i] < 0){
                swap(dpmax, dpmin);
            }

            //dpmax、dpmin只记录nums[i]结尾的数组最大以及最小
            dpmax = max(dpmax*nums[i], nums[i]);
            dpmin = min(dpmin* nums[i], nums[i]);

            //寻找最大的dpmax
            ans = max(ans, dpmax);
        }

        return ans;
    }
};
``
