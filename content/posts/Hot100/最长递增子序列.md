1. 本题主要思路就是对状态前放的每个`dp`状态进行遍历，寻找当前状态最大的dp可能性
2. 注意的是返回的答案是dp数组中的最大值，而不是`dp[n-1]`
```c++
class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        size_t n = nums.size();
        vector<int> dp(n, 1);

        int ans = 0;
        for(int i = 0; i < n; i++){
            for(int j = 0; j < i; j++){
                if(nums[i] > nums[j]){
                    dp[i] = max(dp[i], dp[j]+1);
                }
            }
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
```
