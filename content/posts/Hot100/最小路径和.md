1. 最小路径和求加权最小值，状态转移方程是取最小
```c++
class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        size_t m = grid.size();
        size_t n = grid[0].size();
        vector<vector<int>> dp(m, vector<int>(n, 0));
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                int minnum = INT_MAX;
                if(i-1 >= 0){
                    minnum = min(minnum, dp[i-1][j]);
                }
                if(j-1 >= 0){
                    minnum = min(minnum, dp[i][j-1]);
                }
                minnum = minnum == INT_MAX ? 0 : minnum;
                dp[i][j] = minnum + grid[i][j];
            }
        }
        return dp[m-1][n-1];
    }
};
```
