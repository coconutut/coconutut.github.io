1. 背包问题：
    * 01背包问题 -> 倒序遍历容量
    * 完全背包问题 -> 正序遍历容量
2. 物品与容量嵌套顺序
    * 关心排序顺序 -> 容量 + 物品
    * 不关心排序顺序 -> 物品 + 容量（常用这个）

```c++
class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int target = 0;
        for(int num : nums){
            target += num;
        }

        if(target % 2) return false;
        else target /= 2;

        vector<bool> dp(target+1, false);
        dp[0] = true;

        for(int num : nums){
            for(int i = target; i > 0; i--){
                if(i >= num){
                    dp[i] = dp[i-num] || dp[i];
                }
            }
        }

        return dp[target];
    }
};
```
