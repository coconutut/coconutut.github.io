```c++
class Solution {
public:
    int minDistance(string word1, string word2) {
        int m = word1.length();
        int n = word2.length();

        if(m*n == 0) return m+n;

        vector<vector<int>> dp(m+1, vector<int>(n+1, 0));

        for(int i = 0; i <= m; i++){
            dp[i][0] = i;
        }
        for(int i = 0; i <= n; i++){
            dp[0][i] = i;
        }

        for(int i = 1; i <= m; i++){
            for(int j = 1; j <= n; j++){
                //相当于word1去除一个字符 + (i-1, j)状态的操作数
                int delete_ = dp[i-1][j] + 1;

                //相当于word2去除一个字符 + (i, j-1)状态的操作数
                int insert_ = dp[i][j-1] + 1;

                //相当于在(i-1, j-1)状态下，word1变更一个字符
                int update_ = dp[i-1][j-1];
                if(word1[i-1] == word2[j-1]){
                    //与dp[i-1][j-1]无区别，无需变更
                    update_ = dp[i-1][j-1];
                }else{
                    //变更，操作加一
                    update_++;
                }

                //操作数取最小
                dp[i][j] = min(delete_, min(insert_, update_));
            }
        }
        return dp[m][n];
    }
};
```
