1. 位运算
2. 异或性质：
    * `0 ^ x = x`
    * `x ^ x = 0`
    * `a ^ a ^ b = b ^ a ^ a`
```c++
class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int ans = 0;
        for(int n : nums){
            ans ^= n;
        }
        return ans;
    }
};
```
