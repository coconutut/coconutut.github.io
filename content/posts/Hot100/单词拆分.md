1. 本题是比较经典的题目，我们一贯定义`dp[n]->bool`表示以第n个单词结尾时候，能否由wordDict组合；
2. 注意的是转移状态是只要组成即可，所以当dp[i]一旦为true时候，则为true；
```c++
class Solution {
public:
    bool wordBreak(string s, vector<string>& wordDict) {
        size_t n = s.size();
        vector<bool> dp(n+1, false);
        dp[0] = true;

        for(int i = 1; i <= n; i++){
            for(int j = 0; j < wordDict.size(); j++){
                size_t len = wordDict[j].size();
                if(i >= len){
                    bool flag = true;
                    for(int k = 0; k < len; k++){
                        if(s[i-len+k] != wordDict[j][k]){
                            flag = false;
                            break;
                        }
                    }
                    dp[i] = (dp[i-len] && flag);
                }
            }
        }

        return dp[n];
    }
};
```
